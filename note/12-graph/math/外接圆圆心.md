# 基础方案

## 一、问题建模

已知三点：

$$
A(x_1,y_1),\quad B(x_2,y_2),\quad C(x_3,y_3)
$$

它们位于 **xOy 平面**且不共线。

外接圆圆心设为：

$$
O(x,y)
$$

根据外接圆定义：

$$
OA = OB = OC
$$

因此可以得到：

$$
(x-x_1)^2+(y-y_1)^2=(x-x_2)^2+(y-y_2)^2
$$

$$
(x-x_1)^2+(y-y_1)^2=(x-x_3)^2+(y-y_3)^2
$$

---

## 二、展开并消去二次项

将第一式展开：

$$
(x-x_1)^2+(y-y_1)^2-(x-x_2)^2-(y-y_2)^2=0
$$

展开得到：

$$
x^2-2xx_1+x_1^2+y^2-2yy_1+y_1^2
-
(x^2-2xx_2+x_2^2+y^2-2yy_2+y_2^2)=0
$$

消去 \(x^2\) 与 \(y^2\) 后整理得到：

$$
2(x_2-x_1)x + 2(y_2-y_1)y =
x_2^2+y_2^2-x_1^2-y_1^2
$$

同理第二式可得：

$$
2(x_3-x_1)x + 2(y_3-y_1)y =
x_3^2+y_3^2-x_1^2-y_1^2
$$

这样就得到了一个 **二维线性方程组**。

---

## 三、线性方程组形式

$$
\begin{cases}
2(x_2-x_1)x + 2(y_2-y_1)y = x_2^2+y_2^2-x_1^2-y_1^2 \\
2(x_3-x_1)x + 2(y_3-y_1)y = x_3^2+y_3^2-x_1^2-y_1^2
\end{cases}
$$

写成矩阵形式：

$$
\begin{bmatrix}
2(x_2-x_1) & 2(y_2-y_1) \\
2(x_3-x_1) & 2(y_3-y_1)
\end{bmatrix}
\begin{bmatrix}
x\\y
\end{bmatrix}
=
\begin{bmatrix}
x_2^2+y_2^2-x_1^2-y_1^2 \\
x_3^2+y_3^2-x_1^2-y_1^2
\end{bmatrix}
$$

---

## 四、直接公式解

令
$$
A = x_2 - x_1\\
B = y_2 - y_1\\
C = x_3 - x_1\\
D = y_3 - y_1\\
E = x_2^2 + y_2^2 - x_1^2 - y_1^2\\
F = x_3^2 + y_3^2 - x_1^2 - y_1^2\\
$$

行列式：

$$
\Delta = 2(AD - BC)
$$

若：

$$
\Delta = 0
$$

说明三点共线，不存在唯一外接圆。

圆心坐标为：

$$
x = \frac{DE - BF}{2(AD-BC)}
$$

$$
y = \frac{AF - CE}{2(AD-BC)}
$$

## 五、Three.js 实现

函数参数为 **Three.js 的 Vector3**，假设三点 **z 相同（位于 xOy 平面）**。
已经测试

```javascript
import * as THREE from 'three'
/**
 * 计算三点外接圆圆心
 * @param {THREE.Vector3} p1
 * @param {THREE.Vector3} p2
 * @param {THREE.Vector3} p3
 * @returns {THREE.Vector3|null}
 */
function circumcenter(p1, p2, p3) {

    const x1 = p1.x, y1 = p1.y
    const x2 = p2.x, y2 = p2.y
    const x3 = p3.x, y3 = p3.y

    const A = x2 - x1
    const B = y2 - y1
    const C = x3 - x1
    const D = y3 - y1

    const E = x2*x2 + y2*y2 - x1*x1 - y1*y1
    const F = x3*x3 + y3*y3 - x1*x1 - y1*y1

    const det = 2 * (A * D - B * C)

    if (Math.abs(det) < 1e-10) {
        // 三点共线
        return null
    }

    const x = (D * E - B * F) / det
    const y = (A * F - C * E) / det

    return new THREE.Vector3(x, y, 0)
}
```

# 另一种方案：

$$
O=A+\frac{|\vec{AB}|}{2sin(∠BAC)}i_{AC_⊥}-\frac{|\vec{AC}|}{2sin(∠BAC)}j_{AB_⊥}\\
O=A+\frac{|\vec{AB}|^2\vec{AC}_⊥}{2|\vec{AB}×\vec{AC}|}-\frac{|\vec{AC}|^2\vec{AB}_⊥}{2|\vec{AB}×\vec{AC}|}
$$

解释：

从$A$点出发，沿着$\vec{AC}_⊥$ 方向前进$\frac{|\vec{AB}|}{2sin(∠BAC)}$ ，即$|\vec{AB}|$ 的一半为直角边，对角大小等于∠BAC的直角三角形的斜边长；再沿着$-\vec{AB}_⊥ $ 方向前进$\frac{|\vec{AC}|}{2sin(∠BAC)}$ ，即$|\vec{AC}|$ 的一半为直角边，对角大小等于∠BAC的直角三角形的斜边长；

```js
// xOy平面内计算外接圆
function threePointCenter(A, B, C) {
  // 1. 计算相对于 A 点的向量
  const ax = B.x - A.x;
  const ay = B.y - A.y;
  const bx = C.x - A.x;
  const by = C.y - A.y;

  // 2. 计算分母 (2倍的向量叉积)
  const d = 2 * (ax * by - ay * bx);

  // 3. 检查三点是否共线
  if (Math.abs(d) < 1e-10) {
    console.error("三点共线，无法形成圆");
    return null; 
  }

  // 4. 计算向量的平方长度
  const aLen2 = ax * ax + ay * ay;
  const bLen2 = bx * bx + by * by;

  // 5. 根据克拉默法则计算圆心偏移量 (注意这里的对应关系)
  // (by, -bx)是AC逆时针旋转90度  (ay, -ax)是AB逆时针旋转90度
  const offsetX = (by * aLen2 - ay * bLen2) / d;
  const offsetY = (-bx * aLen2 - (-ax * bLen2)) / d;

  // 6. 加上 A 点坐标还原到世界坐标系
  // 保持 z 轴与输入点一致（假设在同一平面）
  return new THREE.Vector3(A.x + offsetX, A.y + offsetY, 0);
}
```

```js
/**
 * 计算空间中任意三点的外接圆圆心 (3D 通用版)
 * @param {THREE.Vector3} A
 * @param {THREE.Vector3} B
 * @param {THREE.Vector3} C
 * @returns {THREE.Vector3 | null}
 */
function getThreePointCenter3D(A, B, C) {
  // 1. 创建基向量
  const u = new THREE.Vector3().subVectors(B, A);
  const v = new THREE.Vector3().subVectors(C, A);
  
  // 2. 计算法向量 n = u x v
  const n = new THREE.Vector3().crossVectors(u, v);
  const nLenSq = n.lengthSq();

  // 3. 检查是否共线 (法向量长度接近0)
  if (nLenSq < 1e-12) {
    console.warn("Points are collinear, no circumcenter exists.");
    return null;
  }

  // 4. 根据公式计算相对偏移向量 p
  // p = ((|u|^2 * v - |v|^2 * u) x n) / (2 * |n|^2)
  
  const uLenSq = u.lengthSq();
  const vLenSq = v.lengthSq();

  // term1 = |u|^2 * v
  const term1 = v.clone().multiplyScalar(uLenSq);
  // term2 = |v|^2 * u
  const term2 = u.clone().multiplyScalar(vLenSq);

  // 重点：计算 (term1 - term2) x n
  const p = new THREE.Vector3()
    .subVectors(term1, term2)
    .cross(n)
    .divideScalar(2 * nLenSq);

  // 5. 最终圆心 = A + p
  return new THREE.Vector3().addVectors(A, p);
}
```
